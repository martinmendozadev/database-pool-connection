git init				--->  Iniiar un poryecto
git reset --soft HEAD-1			--->  Deshacer el último commit
git add NombreArhivo			--->  Agrega al commit un archivo en especifico
git commit -am "TituloCommit"		--->  Agrega archivos modificados y solo pone titulo al commit
git add . && commit -m "Titulo"		--->  Hace el add y el commit 
git log					--->  Muestra una lista de tus commits hechos
git branch				--->  Ver todas tus ramas
git checkout NombreRama			--->  Cambiar de rama
git checkout -b NombreRama		--->  Crea la rama y te cambia
git log --graph --oneline --decorate	--->  Muestra los commits, pocos detalles, en una linea (head)
git log --oneline --color | cat		--->  Muestra id de commits y titulo en lista
git merge NombreRama			--->  Trae tus commits a la rama actual
git merge --abort			--->  Cancela el merge
git diff 				--->  Muestra las lineas que has modificado
git diff IDcommit IDcommit		--->  Muestra las lineas que has modificado entr dos commits
git branch -d NombreRama		--->  Elimina la rama, compurbea que este integrada la rama
git branch -D NombreRama		--->  Elimina la rama, no compruebe que este integrada la rama
git branch -m NombreRama NuevoNombre	--->  Renombra una rama
git gc					--->  Recolector de basura
git commit --amend			--->  Editar cualquier cosa del commit que acabas de hacer (Usalo sabiamente)
git checkout -- file.ex			--->  Regresas al estado de inicio del commit
git reset IDcommit			--->  Eliminar último commit (7 bits id, despues checkout -- file)
git revert IDCommit IDCommit 		--->  Descahace cambios (hace commit)
git revert --no-commit IDCommit		--->  Deshace los cambions pero no crea un commit, para que puedas peter otro revert
git show IDCommit			--->  Muestra descripcion del commit
git stash				--->  Guarda tu avance en un buffer, cambiar rama si commit (pila)
git stash aplly				--->  Regresa tu avance
git stash drop				--->  Elimina stash
git stash save "Titulo"			--->  Stash con nombre
git stash show stash@\{0\}		--->  Muestas tus stash
git stash pop				--->  Saca el ultimo stahs
git stash branch NombreRama Stash	--->  Combierte tu stash en una rama
git log --online --decorate --graph --all --since=2019-09-21 ---> Muetras tus commits de una fecha en espefifico a la fecha actual
git rebase NombreRama			--->  Ordena tus commits en una rama, limpia historial (cuidado es destructivo)
git rebase -i HEAD~4			--->  Modifica grapho (cuidado)
